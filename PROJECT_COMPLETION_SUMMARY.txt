💡 EXAMPLE USAGE
═══════════════════════════════════════════════════════════════════════════════

# Initialize
from weather_aggregator import WeatherAggregator
agg = WeatherAggregator(cache_ttl_minutes=30)

# Get complete data
weather = agg.get_complete_weather_data(
    latitude=40.7128,
    longitude=-74.0060,
    location_name="NYC",
    forecast_days=7,
    station_code="KJFK"
)

# Extract ML features
from weather_utils import WeatherFeatureExtractor
features = {
    **WeatherFeatureExtractor.current_conditions_dict(weather),
    **WeatherFeatureExtractor.hourly_statistics(weather.hourly_forecast, 24),
}

# Market calculations
from weather_utils import WeatherAggregations
temp_prob = WeatherAggregations.temperature_exceedance_probability(
    weather.hourly_forecast, threshold=15, hours_ahead=24
)
print(f"Probability of exceeding 15°C: {temp_prob:.1%}")

# Make prediction
prediction = your_model.predict(features)

# Place trade
if prediction.confidence > 0.65:
    kalshi.place_bet(market, prediction)
